SELECT TOP(50)
  WH.WORKORDERID,
	WH.DUEDATE,
	WH.CREATEDBY,
	WH.CREATEDDATETIME,
	PT.NAME AS PROJNAME,
	DPT.NAME AS CUSTNAME
FROM
  dbo.ESTWORKORDERHEADER WH
JOIN PROJTABLE PT ON PT.PROJID = WH.PROJID AND PT.PARTITION = WH.PARTITION AND PT.DATAAREAID = WH.DATAAREAID
JOIN CUSTTABLE CT ON CT.ACCOUNTNUM = PT.CUSTACCOUNT AND CT.PARTITION = WH.PARTITION AND CT.DATAAREAID = WH.DATAAREAID
JOIN DIRPARTYTABLE DPT ON DPT.RECID = CT.PARTY
WHERE WH.DATAAREAID = 'tta' AND WH.PARTITION = 5637144576
ORDER BY DUEDATE DESC
